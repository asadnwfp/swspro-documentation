/**
 * Global iife: Business Area
 * @class BusinessArea
 * @category Forms
 */
(function () {
  var strNames;
  wpw.tax.create.calcBlocks("businessarea", function (calcUtils) {
    calcUtils.onFormLoad("businessarea", function (calcUtils) {
      // set the tab index to 1
      calcUtils.field("BUSINESSAREA-TAB-INDEX").assign("1");
      calcUtils.field("BUSINESSAREA-TABS").cell(0, 1).config("class", "");
      calcUtils
        .field("BUSINESSAREA-TABS")
        .cell(0, 0)
        .config("class", "de-cards-tabs-new");
      //assign text to einfuhrungs tab
      calcUtils
        .field("BUSINESSAREA-EDITOR")
        .assign(
          "<p>Definieren Sie hier die Geschäftsbereiche der Kommune. Die hier definierten Geschäftsbereiche können sie später den jeweiligen Risiken zuordnen.</p><p>&#160;</p><p>Die Geschäftsbereiche sind lediglich eine informative Angabe. Diese haben keinen direkten Einfluss auf Verantwortlichkeiten. Verantwortliche Personen und Aufgaben werden direkt in den jeweiligen Abfragen definiert.</p><p>&#160;</p><p>Diese Funktion dient der informativen Einteilung von Risiken. Hierdurch kann in den Risikoeinstellungen bereits hinterlegt werden, welche Bereiche potentiell vom jeweiligen Risiko betroffen sind.</p><p>Die Einteilung in Geschäftsbereiche orientiert sich hierbei frei nach der internen Verwaltungsstruktur. So können Dezernate, Fachämter, Verantwortlichkeitsbereiche oder ähnliches hinterlegt werden.</p><p>&#160;</p><p>Zum Beispiel:</p><ul><li>Fachbereich 01: Finanzen</li><li>Hauptamt</li><li>Kämmerei</li><li>Dezernat Bauen und Umwelt</li><li>…</li></ul>"
        );
      disableTable();
    });

    function disableTable() {
      let table = wpw.tax.form("businessarea").field("BUSINESSAREA-TABLE");
      let tableLength = table.getRows().length;
      for (let i = 0; i < tableLength; i++) {
        table.cell(i, 0).disabled(true);
        table.cell(i, 0).config("cannotOverride", true);
      }

      // hide the add remove row buttons
      angular.element(document).ready(() => {
        setTimeout(() => {
          var element = document.getElementById(
            wpw.tax.field("BUSINESSAREA-TABLE").getId()
          );
          element.classList.add("hide-column");
        }, 500);
      });
    }

    calcUtils.custom("btnBA", function (calcUtils) {
      // set the tab index to 1
      calcUtils.field("BUSINESSAREA-TAB-INDEX").assign("1");
      calcUtils.field("BUSINESSAREA-TABS").cell(0, 1).config("class", "");
      calcUtils
        .field("BUSINESSAREA-TABS")
        .cell(0, 0)
        .config("class", "de-cards-tabs-new");
      disableTable();
    });

    calcUtils.custom("btnBADesc", function (calcUtils) {
      // set the tab index to 2
      calcUtils.field("BUSINESSAREA-TAB-INDEX").assign("2");
      calcUtils.field("BUSINESSAREA-TABS").cell(0, 0).config("class", "");
      calcUtils
        .field("BUSINESSAREA-TABS")
        .cell(0, 1)
        .config("class", "de-cards-tabs-new");
    });
    calcUtils.custom("saveBAItems", function (calcUtils) {
      let itemsTable = calcUtils.form("businessarea").field("SELECT-BA-ITEMS");
      let tabLength = itemsTable.getRows().length;
      var names = [];
      for (let i = 0; i < tabLength; i++) {
        let name = itemsTable.cell(i, 0).get();
        let assigned = itemsTable.cell(i, 1).get();

        if (assigned === true) {
          names.push(name);
        }
      }
      strNames = names.join(", ");
      wpw.tax.form("riskcard").field("BA").set(strNames);
      wpw.tax.closeModal();
    });
  });
})();
