    function saveRelevant() {
      wpw.tax.getRisks().then((risks) => {
        risks.forEach((risk) => {
          const riskRelevantStatus = risk.properties.DE_relevant; // contains "Ja" or ""
          const riskActiveStatus = risk.properties.DE_activeStatus; // contains "1" or "0" or initial value as ""

          let clonedRisk = wpw.utilities.clone(risk);
          if (alwaysRelevantRisks.indexOf(risk.id) > -1) {
            if (risk.properties.DE_relevant !== "Ja") {
              clonedRisk.properties.DE_relevant = "Ja";
              clonedRisk.properties.DE_activeStatus = "1";
            }
          } else {
            if (relevantRisk(risk.id)) {
              //Four states when a risk need to be relevant
              //relevant	active		        relevant	active
              //relevant	not active		    relevant	not active
              //not relevant	active		    relevant	active
              //not relevant	not active		relevant	active

              if (riskRelevantStatus !== "Ja" && riskActiveStatus === "1") {
                clonedRisk.properties.DE_relevant = "Ja";
              } else if (
                riskRelevantStatus !== "Ja" &&
                riskActiveStatus !== "1"
              ) {
                clonedRisk.properties.DE_relevant = "Ja";
                clonedRisk.properties.DE_activeStatus = "1";
              }
            } else {
              // four states when a risk need to be not relevant
              //  relevant	active		      not relevant	active
              // relevant	not active		    not relevant	not active
              // not relevant	active		    not relevant	active
              // not relevant	not active		not relevant	not active

              if (riskRelevantStatus === "Ja" && riskActiveStatus === "1") {
                clonedRisk.properties.DE_relevant = "";
              } else if (
                riskRelevantStatus === "Ja" &&
                riskActiveStatus !== "1"
              ) {
                clonedRisk.properties.DE_relevant = "";
                clonedRisk.properties.DE_activeStatus = "0";
              }
            }
          }
          return wpw.tax.saveRisk(clonedRisk);
        });
      });
    }
