/**
 * Global iife: Select Documents
 * @class SelectDocuments
 * @category Forms
 */
(function () {
  /* global _ */
  const gCalcs = wpw.tax.de.global.functions.gCalcs;
  wpw.tax.create.calcBlocks("selectdocuments", function (calcUtils) {
    calcUtils.onFormLoad("selectdocuments", function (calcUtils) {
      const guid = calcUtils.form("riskcard").field("GUID").get();

      // set the tab index to 1
      calcUtils.field("RISKDOCUMENTS-TAB-INDEX").assign("1");
      calcUtils.field("RISKDOCUMENTS-TABS").cell(0, 1).config("class", "");
      calcUtils
        .field("RISKDOCUMENTS-TABS")
        .cell(0, 0)
        .config("class", "de-cards-tabs-new");
      //assign text to einfuhrungs tab
      calcUtils
        .field("RISKDOCUMENTS-EDITOR")
        .assign(
          'Dokumente, welche sie zuordnen möchten, müssen Sie zuerst dem Projekt hinzufügen. Gehen Sie hierzu in die Dokumentenübersicht (Registerkarte "Dokumente"). Entscheiden Sie sich nun für einen Ordner, in welchem Sie das Dokument ablegen möchten. Wir empfehlen Ihnen hierzu den Ordner "Bibliothek" innerhalb des Bereichs "Dokumente". Ziehen Sie nun das Dokument, welches Sie hochladen möchten, per Drag&Drop in den ausgewählen Ordner. Gehen Sie abschließend zurück in diesen Dialog und das hinzugefügte Dokument steht Ihnen nun zur Zuordnung zur Verfügung.'
        );
      return gCalcs.extractDocuments(
        calcUtils,
        "selectdocuments",
        "SELECT-DOCUMENTS",
        guid
      );
    });

    calcUtils.custom("btnRiskDocuments", function (calcUtils) {
      // set the tab index to 1
      calcUtils.field("RISKDOCUMENTS-TAB-INDEX").assign("1");
      calcUtils.field("RISKDOCUMENTS-TABS").cell(0, 1).config("class", "");
      calcUtils
        .field("RISKDOCUMENTS-TABS")
        .cell(0, 0)
        .config("class", "de-cards-tabs-new");
    });

    calcUtils.custom("btnRiskDocumentsDescription", function (calcUtils) {
      // set the tab index to 2
      calcUtils.field("RISKDOCUMENTS-TAB-INDEX").assign("2");
      calcUtils.field("RISKDOCUMENTS-TABS").cell(0, 0).config("class", "");
      calcUtils
        .field("RISKDOCUMENTS-TABS")
        .cell(0, 1)
        .config("class", "de-cards-tabs-new");
    });

    calcUtils.custom("saveAssignedDocuments", function (calcUtils) {
      let tableRows = calcUtils.field("SELECT-DOCUMENTS").getRows();
      const guid = calcUtils.form("riskcard").field("GUID").get();
      return wpw.tax.getRisks().then((risks) => {
        const risk = risks.find((risk) => risk.id === guid);
        if (!risk.properties.hasOwnProperty("DE_assignedDocuments")) {
          risk.properties.DE_assignedDocuments = "";
        }
        const clonedRisk = wpw.utilities.clone(risk);
        let docIds = [];
        if (clonedRisk.properties.DE_assignedDocuments !== "") {
          docIds = clonedRisk.properties.DE_assignedDocuments.split(",");
        }
        for (let i = 0; i < tableRows.length; i++) {
          const status = tableRows[i][1].get();
          const dId = tableRows[i][2].get();
          // if the checkbox is checked this document id need to be saved
          if (status === true) {
            if (!docIds.includes(dId)) {
              docIds.push(dId);
            }
          }
          // if the checkbox is not checked or it is chnaged from previously checked to uncheck
          // we need to remove that id if exists already
          else {
            const index = docIds.indexOf(dId);
            if (index > -1) {
              docIds.splice(index, 1);
            }
          }
          clonedRisk.properties.DE_assignedDocuments = docIds.join();
        }
        return wpw.tax.saveRisk(clonedRisk).then((res) => {
          console.log("saved risk documents");
          // recall the table
          gCalcs.generateEngDocs(calcUtils, clonedRisk, "riskcard", "E-DOCS");
          wpw.tax.closeModal();
        });
      });
    });
  });
})();
