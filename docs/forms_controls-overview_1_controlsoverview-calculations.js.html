/**
 * Global iife: Controls Overview
 * @class ControlsOverview
 * @category Forms
 */
(function () {
  /* global _ */
  const gCalcs = wpw.tax.de.global.functions.gCalcs;
  wpw.tax.create.calcBlocks("controlsoverview", function (calcUtils) {
    calcUtils.onFormLoad("controlsoverview", function (calcUtils) {
      // this will be run from the calculations to trigger the bar
      gCalcs.startLoadingBar(
        "CONTROLSOVERVIEW",
        "Bitte warten",
        "Maßnahmenübersicht wird geladen. Bitte warten."
      );
      // set the tab index to 1
      calcUtils.field("CONTROLSOVERVIEW-TAB-INDEX").assign("1");
      calcUtils
        .field("CONTROLSSOVERVIEW-TABS")
        .cell(0, 0)
        .config("class", "de-tabs-new");
      // assign the text to einfurung tab
      calcUtils
        .field("CONTROLS-OVERVIEW-EDITOR")
        .assign(
          '<p>Dieses Dokument gibt Ihnen eine Übersicht über die Maßnahmen, die Sie den Risiken zugewiesen haben. Sie können weitere individuelle Maßnahmen über die Schaltfläche "Maßnahme anlegen" hinzufügen.</br></br>Klicken Sie auf den Stift in der Spalte "Details“, um die "Maßnahmenkarte" zu öffnen, in der weitere Informationen zur Maßnahme angezeigt werden. Hier können Sie ggfs. Anpassungen vornehmen.</br></br>Überprüfen Sie in der Spalte "Risiko zugeordnet", ob alle Maßnahmen einem Risiko zugeordnet sind. Sollte dies nicht der Fall sein, gehen Sie zurück ins Dokument "Risikoübersicht", wählen Sie das entsprechende Risiko aus und nehmen Sie eine Zuordnung vor. Sollte die Maßnahme überflüssig sein, setzen Sie Standardmaßnahmen in der Maßnahmenkarte inaktiv, bzw. löschen Sie individuelle Maßnahmen.</p>'
        );
      return gCalcs.extractControls(calcUtils);
    });

    calcUtils.custom("btnControlsOverview", function (calcUtils) {
      // set the tab index to 1
      calcUtils.field("CONTROLSOVERVIEW-TAB-INDEX").assign("1");
      calcUtils.field("CONTROLSSOVERVIEW-TABS").cell(0, 1).config("class", "");
      calcUtils
        .field("CONTROLSSOVERVIEW-TABS")
        .cell(0, 0)
        .config("class", "de-tabs-new");
    });

    calcUtils.custom("btnControlsDescription", function (calcUtils) {
      // set the tab index to 2
      calcUtils.field("CONTROLSOVERVIEW-TAB-INDEX").assign("2");
      calcUtils.field("CONTROLSSOVERVIEW-TABS").cell(0, 0).config("class", "");
      calcUtils
        .field("CONTROLSSOVERVIEW-TABS")
        .cell(0, 1)
        .config("class", "de-tabs-new");
    });

    calcUtils.custom("editControlsRow", function (calcUtils) {
      wpw.tax.resetForm("controlcard");
      return gCalcs.openControlCard("CONTROLS-OVERVIEW", calcUtils);
    });

    calcUtils.custom("editEControlsRow", function (calcUtils) {
      wpw.tax.resetForm("controlcard");
      return gCalcs.openControlCard("ECONTROLS-OVERVIEW", calcUtils);
    });

    calcUtils.custom("addNewControl", function (calcUtils) {
      wpw.tax.resetForm("controlcard");
      wpw.tax.openModal("controlcard");
    });

    calcUtils.custom("controlsOverviewHelp", function () {
      const url =
        "https://docu.audicon.net/CWCloud/Apps/SWS.pro/de/Content/Massnahmenuebersicht.htm";
      // self for same window, blank for new window
      window.open(url, "_blank");
    });
  });
})();
